00059 bool isFinalB(int n, const Event& evt) {
00060   // *This* particle must be a b or b-hadron

00081 vector<int> get_anaparticles(const Event& evt) {

00083   vector<int> unstables;
00084   for (int n = 0; n < evt.size(); ++n) {

00087   }
00088   // Identify the final state particles from those decays
00089   vector<int> rmparticles;
00090   for (int n : unstables) finalDescendants(n, evt, rmparticles);
00091   // Identify all final state particles, except those from b and tau decays where the parents are used
00092   vector<int> rtn = unstables;

00094     if (!evt[n].isFinal()) continue;
00095     if (find(rmparticles.begin(), rmparticles.end(), n) == rmparticles.end()) rtn.push_back(n);
00096   }
00097   sort(rtn.begin(), rtn.end());
