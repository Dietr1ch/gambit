00002 /* -------------------------------------------------------------------------
00003  * This is an ANSI C library for multi-stream random number generation.  
00004  * The use of this library is recommended as a replacement for the ANSI C 
00005  * rand() and srand() functions, particularly in simulation applications 
00006  * where the statistical 'goodness' of the random number generator is 
00007  * important.  The library supplies 256 streams of random numbers; use 
00008  * SelectStream(s) to switch between streams indexed s = 0,1,...,255.
00009  *
00010  * The streams must be initialized.  The recommended way to do this is by
00011  * using the function PlantSeeds(x) with the value of x used to initialize 
00012  * the default stream and all other streams initialized automatically with
00013  * values dependent on the value of x.  The following convention is used 
00014  * to initialize the default stream:
00015  *    if x > 0 then x is the state
00016  *    if x < 0 then the state is obtained from the system clock
00017  *    if x = 0 then the state is to be supplied interactively.
00018  *
00019  * The generator used in this library is a so-called 'Lehmer random number
00020  * generator' which returns a pseudo-random number uniformly distributed
00021  * 0.0 and 1.0.  The period is (m - 1) where m = 2,147,483,647 and the
00022  * smallest and largest possible values are (1 / m) and 1 - (1 / m)
00023  * respectively.  For more details see:
00024  * 
00025  *       "Random Number Generators: Good Ones Are Hard To Find"
00026  *                   Steve Park and Keith Miller
00027  *              Communications of the ACM, October 1988
00028  *
00029  * Name            : rngs.c  (Random Number Generation - Multiple Streams)
00030  * Authors         : Steve Park & Dave Geyer
00031  * Language        : ANSI C
00032  * Latest Revision : 09-22-98
00033  * ------------------------------------------------------------------------- 
00034  */

00052    double Random(void)
00053 /* ----------------------------------------------------------------
00054  * Random returns a pseudo-random real number uniformly distributed 
00055  * between 0.0 and 1.0. 
00056  * ----------------------------------------------------------------
00057  */
00058 {
00059   const long Q = MODULUS / MULTIPLIER;
00060   const long R = MODULUS % MULTIPLIER;
00061         long t;
00062 
00063   t = MULTIPLIER * (seed[stream] % Q) - R * (seed[stream] / Q);
00064   if (t > 0) 
00065     seed[stream] = t;
00066   else 
00067     seed[stream] = t + MODULUS;
00068   return ((double) seed[stream] / MODULUS);
00069 }

00072    void PlantSeeds(long x)
00073 /* ---------------------------------------------------------------------
00074  * Use this function to set the state of all the random number generator 
00075  * streams by "planting" a sequence of states (seeds), one per stream, 
00076  * with all states dictated by the state of the default stream. 
00077  * The sequence of planted states is separated one from the next by 
00078  * 8,367,782 calls to Random().
00079  * ---------------------------------------------------------------------
00080  */
00081 {
00082   const long Q = MODULUS / A256;
00083   const long R = MODULUS % A256;
00084         int  j;
00085         int  s;
00086 
00087   initialized = 1;
00088   s = stream;                            /* remember the current stream */
00089   SelectStream(0);                       /* change to stream 0          */
00090   PutSeed(x);                            /* set seed[0]                 */
00091   stream = s;                            /* reset the current stream    */
00092   for (j = 1; j < STREAMS; j++) {
00093     x = A256 * (seed[j - 1] % Q) - R * (seed[j - 1] / Q);
00094     if (x > 0)
00095       seed[j] = x;
00096     else
00097       seed[j] = x + MODULUS;
00098    }
00099 }
